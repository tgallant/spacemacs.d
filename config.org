** Magit
*** Undo last commit
#+begin_src emacs-lisp
  (defun undo-last-commit ()
    (interactive)
    (magit-reset-soft "HEAD^"))

  (transient-append-suffix 'magit-reset "w"
    '("u" "undo" undo-last-commit))
#+end_src
